//! Basic CPUID Information.
//!
//! When CPUID executes with EAX set to 01H, additional information is
//! returned to the EBX register: • Brand index (low byte of EBX) — this
//! number provides an entry into a brand string table that contains brand
//! strings for IA-32 processors. More information about this field is
//! provided later in this section. • CLFLUSH instruction cache line size
//! (second byte of EBX) — this number indicates the size of the cache line
//! flushed by the CLFLUSH and CLFLUSHOPT instructions in 8-byte increments.
//! This field was introduced in the Pentium 4 processor. • Local APIC ID
//! (high byte of EBX) — this number is the 8-bit ID that is assigned to the
//! local APIC on the processor during power up. This field was introduced
//! in the Pentium 4 processor
use crate::instruction;

#[allow(dead_code)]
pub const NAME: &str  = "leaf_01_ebx";
#[allow(dead_code)]
pub const LONG_NAME: &str  = "Basic CPUID Information";

/// Returns the current value of the register.
///
/// When CPUID executes with EAX set to 01H, additional information is
/// returned to the EBX register: • Brand index (low byte of EBX) — this
/// number provides an entry into a brand string table that contains brand
/// strings for IA-32 processors. More information about this field is
/// provided later in this section. • CLFLUSH instruction cache line size
/// (second byte of EBX) — this number indicates the size of the cache line
/// flushed by the CLFLUSH and CLFLUSHOPT instructions in 8-byte increments.
/// This field was introduced in the Pentium 4 processor. • Local APIC ID
/// (high byte of EBX) — this number is the 8-bit ID that is assigned to the
/// local APIC on the processor during power up. This field was introduced
/// in the Pentium 4 processor
pub fn get() -> u32  {
    let (_eax, ebx, _ecx, _edx) = instruction::execute_cpuid(1,0);
    ebx
}

// Fieldset valid on latest version of the Intel architecture
#[allow(dead_code)]
const BRAND_INDEX_NAME: &str = "brand_index";
#[allow(dead_code)]
const BRAND_INDEX_LONG_NAME: &str = "Brand index";
#[allow(dead_code)]
const BRAND_INDEX_LSB: u32 = 0;
#[allow(dead_code)]
const BRAND_INDEX_MSB: u32 = 7;
#[allow(dead_code)]
const BRAND_INDEX_MASK: u32 = 0x000000ff;

/// Returns the current value of "Brand index".
///
/// Brand index (low byte of EBX) — this number provides an entry into a
/// brand string table that contains brand strings for IA-32 processors.
/// More information about this field is provided later in this section.
pub fn get_brand_index() -> u32  {
    let value: u32 = get();
    (value & BRAND_INDEX_MASK) >> BRAND_INDEX_LSB
}

/// Returns the value of "Brand index" in the given register value.
///
/// Brand index (low byte of EBX) — this number provides an entry into a
/// brand string table that contains brand strings for IA-32 processors.
/// More information about this field is provided later in this section.
pub fn get_brand_index_from_value(value: u32) -> u32  {
    (value & BRAND_INDEX_MASK) >> BRAND_INDEX_LSB
}

#[allow(dead_code)]

/// CLFLUSH instruction cache line size (second byte of EBX) — this number

/// CLFLUSHOPT instructions in 8-byte increments. This field was introduced

pub fn get_clflush_line_size() -> u32  {

}

}

#[allow(dead_code)]
const MAX_ADDRESSABLE_IDS_NAME: &str = "max_addressable_ids";
#[allow(dead_code)]
const MAX_ADDRESSABLE_IDS_LONG_NAME: &str = "Maximum number of addressable IDs";
#[allow(dead_code)]
const MAX_ADDRESSABLE_IDS_LSB: u32 = 16;
#[allow(dead_code)]
const MAX_ADDRESSABLE_IDS_MSB: u32 = 23;
#[allow(dead_code)]
const MAX_ADDRESSABLE_IDS_MASK: u32 = 0x00ff0000;

/// Returns the current value of "Maximum number of addressable IDs".
///
/// Maximum number of addressable IDs for logical processors in this
/// physical package. The nearest power-of-2 integer that is not smaller
/// than EBX\[23:16\] is the number of unique initial APIC IDs reserved for
/// addressing different logical processors in a physical package. This
/// field is only valid if CPUID.1.EDX.HTT\[bit 28\]= 1.
pub fn get_max_addressable_ids() -> u32  {
    let value: u32 = get();
    (value & MAX_ADDRESSABLE_IDS_MASK) >> MAX_ADDRESSABLE_IDS_LSB
}

/// Returns the value of "Maximum number of addressable IDs" in the given register value.
///
/// Maximum number of addressable IDs for logical processors in this
/// physical package. The nearest power-of-2 integer that is not smaller
/// than EBX\[23:16\] is the number of unique initial APIC IDs reserved for
/// addressing different logical processors in a physical package. This
/// field is only valid if CPUID.1.EDX.HTT\[bit 28\]= 1.
pub fn get_max_addressable_ids_from_value(value: u32) -> u32  {
    (value & MAX_ADDRESSABLE_IDS_MASK) >> MAX_ADDRESSABLE_IDS_LSB
}

#[allow(dead_code)]
const INITIAL_APIC_ID_NAME: &str = "initial_apic_id";
#[allow(dead_code)]
const INITIAL_APIC_ID_LONG_NAME: &str = "Initial APIC ID";
#[allow(dead_code)]
const INITIAL_APIC_ID_LSB: u32 = 24;
#[allow(dead_code)]
const INITIAL_APIC_ID_MSB: u32 = 31;
#[allow(dead_code)]
const INITIAL_APIC_ID_MASK: u32 = 0xff000000;

/// Returns the current value of "Initial APIC ID".
///
/// Local APIC ID (high byte of EBX) — this number is the 8-bit ID that is
/// assigned to the local APIC on the processor during power up. This field
/// was introduced in the Pentium 4 processor. The 8-bit initial APIC ID in
/// EBX\[31:24\] is replaced by the 32-bit x2APIC ID, available in Leaf 0BH
/// and Leaf 1FH.
pub fn get_initial_apic_id() -> u32  {
    let value: u32 = get();
    (value & INITIAL_APIC_ID_MASK) >> INITIAL_APIC_ID_LSB
}

/// Returns the value of "Initial APIC ID" in the given register value.
///
/// Local APIC ID (high byte of EBX) — this number is the 8-bit ID that is
/// assigned to the local APIC on the processor during power up. This field
/// was introduced in the Pentium 4 processor. The 8-bit initial APIC ID in
/// EBX\[31:24\] is replaced by the 32-bit x2APIC ID, available in Leaf 0BH
/// and Leaf 1FH.
pub fn get_initial_apic_id_from_value(value: u32) -> u32  {
    (value & INITIAL_APIC_ID_MASK) >> INITIAL_APIC_ID_LSB
}
