//! Deterministic Cache Parameters Leaf.
//!
//! When CPUID executes with EAX set to 04H and ECX contains an index value,
//! the processor returns encoded data that describe a set of deterministic
//! cache parameters (for the cache level associated with the input in ECX).
//! Valid index values start from 0. Software can enumerate the
//! deterministic cache parameters for each level of the cache hierarchy
//! starting with an index value of 0, until the parameters report the value
//! associated with the cache type field is 0. The architecturally defined
//! fields reported by deterministic cache parameters are documented in
//! Table 3-8. This Cache Size in Bytes = (Ways + 1) * (Partitions + 1) *
//! (Line_Size + 1) * (Sets + 1) = (EBX\[31:22\] + 1) * (EBX\[21:12\] + 1) *
//! (EBX\[11:0\] + 1) * (ECX + 1) The CPUID leaf 04H also reports data that
//! can be used to derive the topology of processor cores in a physical
//! package.  This information is constant for all valid index values.
//! Software can query the raw data reported by executing CPUID with EAX=04H
//! and ECX=0 and use it as part of the topology enumeration algorithm
//! described in Chapter 8, “Multiple-Processor Management,” in the Intel®
//! 64 and IA-32 Architectures Software Developer’s Manual, Volume 3A.
use crate::instruction;

#[allow(dead_code)]
pub const NAME: &str  = "leaf_04_edx";
#[allow(dead_code)]
pub const LONG_NAME: &str  = "Deterministic Cache Parameters Leaf";

/// Returns the current value of the register.
///
/// When CPUID executes with EAX set to 04H and ECX contains an index value,
/// the processor returns encoded data that describe a set of deterministic
/// cache parameters (for the cache level associated with the input in ECX).
/// Valid index values start from 0. Software can enumerate the
/// deterministic cache parameters for each level of the cache hierarchy
/// starting with an index value of 0, until the parameters report the value
/// associated with the cache type field is 0. The architecturally defined
/// fields reported by deterministic cache parameters are documented in
/// Table 3-8. This Cache Size in Bytes = (Ways + 1) * (Partitions + 1) *
/// (Line_Size + 1) * (Sets + 1) = (EBX\[31:22\] + 1) * (EBX\[21:12\] + 1) *
/// (EBX\[11:0\] + 1) * (ECX + 1) The CPUID leaf 04H also reports data that
/// can be used to derive the topology of processor cores in a physical
/// package.  This information is constant for all valid index values.
/// Software can query the raw data reported by executing CPUID with EAX=04H
/// and ECX=0 and use it as part of the topology enumeration algorithm
/// described in Chapter 8, “Multiple-Processor Management,” in the Intel®
/// 64 and IA-32 Architectures Software Developer’s Manual, Volume 3A.
pub fn get_at_index(index: usize) -> u32  {
    let (_eax, _ebx, _ecx, edx) = instruction::execute_cpuid(4,index as u32);
    edx
}

// Fieldset valid on latest version of the Intel architecture
#[allow(dead_code)]
const WRITE_BACK_INVALIDATE_INVALIDATE_NAME: &str = "write_back_invalidate_invalidate";
#[allow(dead_code)]
const WRITE_BACK_INVALIDATE_INVALIDATE_LONG_NAME: &str = "Write-Back Invalidate/Invalidate";
#[allow(dead_code)]
const WRITE_BACK_INVALIDATE_INVALIDATE_LSB: u32 = 0;
#[allow(dead_code)]
const WRITE_BACK_INVALIDATE_INVALIDATE_MSB: u32 = 0;
#[allow(dead_code)]
const WRITE_BACK_INVALIDATE_INVALIDATE_MASK: u32 = 0x00000001;

/// Returns whether "Write-Back Invalidate/Invalidate" is currently enabled.
///
/// Bit 00: Write-Back Invalidate/Invalidate. 0 = WBINVD/INVD from threads
/// sharing this cache acts upon lower level caches for threads sharing this
/// cache. 1 = WBINVD/INVD is not guaranteed to act upon lower level caches
/// of non-originating threads sharing this cache
pub fn write_back_invalidate_invalidate_is_enabled_at_index(index: usize) -> bool  {
    let value: u32 = get_at_index(index);
    (value & WRITE_BACK_INVALIDATE_INVALIDATE_MASK) != 0
}

/// Returns whether "Write-Back Invalidate/Invalidate" is enabled in the given register value.
///
/// Bit 00: Write-Back Invalidate/Invalidate. 0 = WBINVD/INVD from threads
/// sharing this cache acts upon lower level caches for threads sharing this
/// cache. 1 = WBINVD/INVD is not guaranteed to act upon lower level caches
/// of non-originating threads sharing this cache
pub fn write_back_invalidate_invalidate_is_enabled_in_value(value: u32) -> bool  {
    (value & WRITE_BACK_INVALIDATE_INVALIDATE_MASK) != 0
}

/// Returns whether "Write-Back Invalidate/Invalidate" is currently disabled.
///
/// Bit 00: Write-Back Invalidate/Invalidate. 0 = WBINVD/INVD from threads
/// sharing this cache acts upon lower level caches for threads sharing this
/// cache. 1 = WBINVD/INVD is not guaranteed to act upon lower level caches
/// of non-originating threads sharing this cache
pub fn write_back_invalidate_invalidate_is_disabled_at_index(index: usize) -> bool  {
    let value: u32 = get_at_index(index);
    (value & WRITE_BACK_INVALIDATE_INVALIDATE_MASK) == 0
}

/// Returns whether "Write-Back Invalidate/Invalidate" is disabled in the given register value.

pub fn write_back_invalidate_invalidate_is_disabled_in_value(value: u32) -> bool  {
