//! Basic CPUID Information.
//!
//! When CPUID executes with EAX set to 02H, the processor returns
//! information about the processor’s internal TLBs, cache and prefetch
//! hardware in the EAX, EBX, ECX, and EDX registers. The information is
//! reported in encoded form and fall into the following categories: • The
//! least-significant byte in register EAX (register AL) will always return
//! 01H. Software should ignore this value and not interpret it as an
//! informational descriptor. • The most significant bit (bit 31) of each
//! register indicates whether the register contains valid information (set
//! to 0) or is reserved (set to 1). • If a register contains valid
//! information, the information is contained in 1 byte descriptors. There
//! are four types of encoding values for the byte descriptor, the encoding
//! type is noted in the second column of Table 3-12. Table 3-12 lists the
//! encoding of these descriptors. Note that the order of descriptors in the
//! EAX, EBX, ECX, and EDX registers is not defined; that is, specific bytes
//! are not designated to contain descriptors for specific cache, prefetch,
//! or TLB types. The descriptors may appear in any order. Note also a
//! processor may report a general descriptor type (FFH) and not report any
//! byte descriptor of “cache type” via CPUID leaf 2.
use crate::instruction;

#[allow(dead_code)]
pub const NAME: &str  = "leaf_02_eax";
#[allow(dead_code)]
pub const LONG_NAME: &str  = "Basic CPUID Information";

/// Returns the current value of the register.
///
/// When CPUID executes with EAX set to 02H, the processor returns
/// information about the processor’s internal TLBs, cache and prefetch
/// hardware in the EAX, EBX, ECX, and EDX registers. The information is
/// reported in encoded form and fall into the following categories: • The
/// least-significant byte in register EAX (register AL) will always return
/// 01H. Software should ignore this value and not interpret it as an
/// informational descriptor. • The most significant bit (bit 31) of each
/// register indicates whether the register contains valid information (set
/// to 0) or is reserved (set to 1). • If a register contains valid
/// information, the information is contained in 1 byte descriptors. There
/// are four types of encoding values for the byte descriptor, the encoding
/// type is noted in the second column of Table 3-12. Table 3-12 lists the
/// encoding of these descriptors. Note that the order of descriptors in the
/// EAX, EBX, ECX, and EDX registers is not defined; that is, specific bytes
/// are not designated to contain descriptors for specific cache, prefetch,
/// or TLB types. The descriptors may appear in any order. Note also a
/// processor may report a general descriptor type (FFH) and not report any
/// byte descriptor of “cache type” via CPUID leaf 2.
pub fn get() -> u32  {
    let (eax, _ebx, _ecx, _edx) = instruction::execute_cpuid(2,0);
    eax
}

// Fieldset valid on latest version of the Intel architecture
#[allow(dead_code)]
const CACHE_AND_TLB_INFORMATION_NAME: &str = "cache_and_tlb_information";
#[allow(dead_code)]
const CACHE_AND_TLB_INFORMATION_LONG_NAME: &str = "Cache and TLB Information";
#[allow(dead_code)]
const CACHE_AND_TLB_INFORMATION_LSB: u32 = 0;
#[allow(dead_code)]
const CACHE_AND_TLB_INFORMATION_MSB: u32 = 31;
#[allow(dead_code)]
const CACHE_AND_TLB_INFORMATION_MASK: u32 = 0xffffffff;

/// Returns the current value of "Cache and TLB Information".
pub fn get_cache_and_tlb_information() -> u32  {
    let value: u32 = get();
    (value & CACHE_AND_TLB_INFORMATION_MASK) >> CACHE_AND_TLB_INFORMATION_LSB
}

/// Returns the value of "Cache and TLB Information" in the given register value.
pub fn get_cache_and_tlb_information_from_value(value: u32) -> u32  {
    (value & CACHE_AND_TLB_INFORMATION_MASK) >> CACHE_AND_TLB_INFORMATION_LSB
}
