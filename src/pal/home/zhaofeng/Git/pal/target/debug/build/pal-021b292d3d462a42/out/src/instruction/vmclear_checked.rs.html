use cty::uint64_t;

// vmclear_checked (Error checked vmclear): This instruction applies to the
// VMCS whose VMCS region resides at thephysical address contained in the
// instruction operand. The instructionensures that VMCS data for that VMCS
// (some of these data may be currentlymaintained on the processor) are copied
// to the VMCS region in memory. Italso initializes parts of the VMCS region
// (for example, it sets thelaunch state of that VMCS to clear). See Chapter
// 24, “Virtual-MachineControl Structures”. The operand of this instruction is
// always 64 bitsand is always in memory. If the operand is the current-VMCS
// pointer, thenthat pointer is made invalid (set to FFFFFFFF_FFFFFFFFH). Note
// that theVMCLEAR instruction might not explicitly write any VMCS data to
// memory;the data may be already resident in memory before the VMCLEAR
// isexecuted.

extern {
    fn pal_execute_vmclear_checked(vmcs_pointer: uint64_t) -> bool;
}

pub fn execute_vmclear_checked(vmcs_pointer: u64) -> bool {
    unsafe { pal_execute_vmclear_checked(vmcs_pointer) }
}
